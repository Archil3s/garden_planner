    }

  List<Bed> _filteredBeds() {
    final query = _safeText(searchQuery).toLowerCase();

    return controller.beds.where((bed) {
      final name = _safeText(bed.name).toLowerCase();
      final zone = _safeText(bed.zone).toLowerCase();
      final status = _safeText(bed.status.name).toLowerCase();

      final cropText = bed.crops
          .map(_safeText)
          .where((crop) => crop.isNotEmpty)
          .join(' ')
          .toLowerCase();

      final matchesQuery =
          query.isEmpty ||
          name.contains(query) ||
          zone.contains(query) ||
          status.contains(query) ||
          cropText.contains(query);

      if (!matchesQuery) return false;

      switch (filterMode) {
        case BedFilterMode.all:
          return true;
        case BedFilterMode.healthy:
          return bed.status == BedStatus.ok;
        case BedFilterMode.attention:
          return bed.status == BedStatus.warning;
        case BedFilterMode.issues:
          return bed.status == BedStatus.bad || bed.status == BedStatus.hold;
        case BedFilterMode.withCrops:
          return bed.crops.isNotEmpty ||
              bed.cropPlacements.isNotEmpty ||
              bed.cropBlocks.isNotEmpty;
      }
    }).toList();
  }

  String _safeText(Object? value) {
    if (value == null) return '';
    return value.toString().trim();
  }

class _CurrentArea extends StatelessWidget {
  const _CurrentArea({
    required this.viewMode,
    required this.controller,
    required this.visibleBeds,
    required this.canvasWidth,
    required this.canvasHeight,
    required this.pixelsPerMeter,
    required this.onOpenBedDetails,
    required this.onOpenMapView,
  });

  final BedViewMode viewMode;
  final GardenController controller;
  final List<Bed> visibleBeds;
  final double canvasWidth;
  final double canvasHeight;
  final double pixelsPerMeter;
  final ValueChanged<Bed> onOpenBedDetails;
  final VoidCallback onOpenMapView;

  @override
  Widget build(BuildContext context) {
    switch (viewMode) {
      case BedViewMode.dashboard:
        return PlannerDashboardOverview(
          beds: controller.beds,
          overlapCount: controller.overlapCount,
          showWeather: true,
          locationName: controller.project.locationName,
          latitude: controller.project.latitude,
          longitude: controller.project.longitude,
        );
      case BedViewMode.seedlings:
        return SeedlingsView(
          seedlings: controller.seedlings,
          beds: controller.beds,
          onAddSeedling: controller.addSeedling,
          onUpdateSeedling: controller.updateSeedling,
          onDeleteSeedling: controller.deleteSeedling,
          onTransplantSeedling: controller.transplantSeedlingToBed,
          onOpenMapView: onOpenMapView,
        );
      case BedViewMode.map:
        return _MapView(
          controller: controller,
          beds: visibleBeds,
          canvasWidth: canvasWidth,
          canvasHeight: canvasHeight,
          pixelsPerMeter: pixelsPerMeter,
          onOpenBedDetails: onOpenBedDetails,
        );
      case BedViewMode.cards:
        return _CardsView(
          beds: visibleBeds,
          onOpenBedDetails: onOpenBedDetails,
        );
    }
  }
}

  void _selectPlantForRow(String cropName) {
    setState(() {
      activePlantName = cropName;
      rowStartGarden = null;
      rowCurrentGarden = null;
      rowBed = null;
    });

  void _placeShortRowAtTap(TapUpDetails details) {
    final cropName = activePlantName;
    if (cropName == null) return;

    final point = _gardenPointFromPixels(details.localPosition);
    final bed = _bedAt(point);

    if (bed == null) {
      ScaffoldMessenger.of(context).showSnackBar(
        const SnackBar(content: Text('Click inside a bed to place a row.')),
      );
      return;
    }

    widget.controller.addCropRowToBedRect(
      bedId: bed.id,
      cropName: cropName,
      startX: point.dx - 1.2,
      startY: point.dy,
      endX: point.dx + 1.2,
      endY: point.dy,
    );

    ScaffoldMessenger.of(context).showSnackBar(
      SnackBar(content: Text('$cropName row added to ${bed.name}.')),
    );
  }

  void _startRowDrawing(DragStartDetails details) {
    final cropName = activePlantName;
    if (cropName == null) return;

    final point = _gardenPointFromPixels(details.localPosition);
    final bed = _bedAt(point);

    if (bed == null) {
      ScaffoldMessenger.of(context).showSnackBar(
        const SnackBar(content: Text('Start the row inside a bed.')),
      );
      return;
    }

    widget.controller.selectBed(bed.id);

    setState(() {
      rowStartGarden = point;
      rowCurrentGarden = point;
      rowBed = bed;
    });
  }

  void _updateRowDrawing(DragUpdateDetails details) {
    if (rowStartGarden == null) return;

    setState(() {
      rowCurrentGarden = _gardenPointFromPixels(details.localPosition);
    });
  }

  void _finishRowDrawing(DragEndDetails details) {
    final cropName = activePlantName;
    final start = rowStartGarden;
    final current = rowCurrentGarden;
    final bed = rowBed;

    if (cropName == null || start == null || current == null || bed == null) {
      _cancelRowDrawing();
      return;
    }

    widget.controller.addCropRowToBedRect(
      bedId: bed.id,
      cropName: cropName,
      startX: start.dx,
      startY: start.dy,
      endX: current.dx,
      endY: current.dy,
    );

    setState(() {
      rowStartGarden = null;
      rowCurrentGarden = null;
      rowBed = null;
    });

    ScaffoldMessenger.of(context).showSnackBar(
      SnackBar(content: Text('$cropName row added to ${bed.name}.')),
    );
  }

  void _cancelRowDrawing() {
    setState(() {
      rowStartGarden = null;
      rowCurrentGarden = null;
      rowBed = null;
    });
  }

  Offset _gardenPointFromPixels(Offset pixelPoint) {
    return Offset(
      pixelPoint.dx / widget.pixelsPerMeter,
      pixelPoint.dy / widget.pixelsPerMeter,
    );
  }

  Bed? _bedAt(Offset gardenPoint) {
    for (final bed in widget.beds.reversed) {
      final insideX =
          gardenPoint.dx >= bed.x && gardenPoint.dx <= bed.x + bed.width;

      final insideY =
          gardenPoint.dy >= bed.y && gardenPoint.dy <= bed.y + bed.height;

      if (insideX && insideY) {
        return bed;
      }
    }

    return null;
  }
}

class _CropRowDraftPainter extends CustomPainter {
  const _CropRowDraftPainter({
    required this.cropName,
    required this.startGarden,
    required this.currentGarden,
    required this.bed,
    required this.pixelsPerMeter,
  });

  final String? cropName;
  final Offset? startGarden;
  final Offset? currentGarden;
  final Bed? bed;
  final double pixelsPerMeter;

  @override
  void paint(Canvas canvas, Size size) {
    final name = cropName;
    final start = startGarden;
    final current = currentGarden;
    final targetBed = bed;

    if (name == null || start == null || current == null || targetBed == null) {
      return;
    }

    final rect = _previewRect(targetBed, start, current);

    final pixelRect = Rect.fromLTWH(
      rect.left * pixelsPerMeter,
      rect.top * pixelsPerMeter,
      rect.width * pixelsPerMeter,
      rect.height * pixelsPerMeter,
    );

    final rrect = RRect.fromRectAndRadius(pixelRect, const Radius.circular(18));

    final fillPaint = Paint()
      ..color = const Color(0xFFD8D8D8).withValues(alpha: 0.72)
      ..style = PaintingStyle.fill;

    final borderPaint = Paint()
      ..color = GardenTheme.good
      ..strokeWidth = 2
      ..style = PaintingStyle.stroke;

    canvas.drawRRect(rrect, fillPaint);
    canvas.drawRRect(rrect, borderPaint);

    final labelPainter = TextPainter(
      text: TextSpan(
        text: name,
        style: const TextStyle(
          color: GardenTheme.ink,
          fontSize: 11,
          fontWeight: FontWeight.w900,
        ),
      ),
      textDirection: TextDirection.ltr,
    )..layout(maxWidth: pixelRect.width - 16);

    final labelRect = Rect.fromCenter(
      center: pixelRect.center,
      width: labelPainter.width + 18,
      height: 25,
    );

    final labelPaint = Paint()
      ..color = GardenTheme.panel.withValues(alpha: 0.95)
      ..style = PaintingStyle.fill;

    canvas.drawRRect(
      RRect.fromRectAndRadius(labelRect, const Radius.circular(8)),
      labelPaint,
    );

    labelPainter.paint(
      canvas,
      Offset(
        labelRect.left + 9,
        labelRect.top + (labelRect.height - labelPainter.height) / 2,
      ),
    );
  }

  Rect _previewRect(Bed bed, Offset start, Offset current) {
    final dx = (current.dx - start.dx).abs();
    final dy = (current.dy - start.dy).abs();
    final horizontal = dx >= dy;

    const thickness = 1.1;
    final centerX = (start.dx + current.dx) / 2;
    final centerY = (start.dy + current.dy) / 2;

    if (horizontal) {
      final width = dx < 0.8 ? 2.4 : dx;
      final left = (centerX - width / 2).clamp(
        bed.x,
        bed.x + bed.width - width,
      );
      final top = (centerY - thickness / 2).clamp(
        bed.y,
        bed.y + bed.height - thickness,
      );

      return Rect.fromLTWH(left, top, width, thickness);
    }

    final height = dy < 0.8 ? 2.4 : dy;
    final left = (centerX - thickness / 2).clamp(
      bed.x,
      bed.x + bed.width - thickness,
    );
    final top = (centerY - height / 2).clamp(
      bed.y,
      bed.y + bed.height - height,
    );

    return Rect.fromLTWH(left, top, thickness, height);
  }

  @override
  bool shouldRepaint(covariant _CropRowDraftPainter oldDelegate) {
    return oldDelegate.cropName != cropName ||
        oldDelegate.startGarden != startGarden ||
        oldDelegate.currentGarden != currentGarden ||
        oldDelegate.bed != bed ||
        oldDelegate.pixelsPerMeter != pixelsPerMeter;
  }
}

    return _plants.where((plant) {

      if (favoritesOnly && !favorites.contains(plant.name)) return false;
      if (seedInventoryOnly && !plant.inSeedInventory) return false;

      if (typeFilter == 'Perennials' && !plant.perennial) return false;

      if (typeFilter != 'All Plants' && typeFilter != 'Perennials') {
        if (plant.type != typeFilter) return false;
      }

      if (query.isNotEmpty) {
        final text = '${plant.name} ${plant.type} ${plant.category}'
            .toLowerCase();

        if (!text.contains(query)) return false;
      }

      return true;
    }).toList();
  }
}

class DraggableCropBlock extends StatefulWidget {
  const DraggableCropBlock({
    super.key,
    required this.bedId,
    required this.block,
    required this.pixelsPerMeter,
    required this.controller,
  });

  final String bedId;
  final CropBlock block;
  final double pixelsPerMeter;
  final GardenController controller;

  @override
  State<DraggableCropBlock> createState() => _DraggableCropBlockState();
}

class _DraggableCropBlockState extends State<DraggableCropBlock> {
  Offset dragOffsetPixels = Offset.zero;
  bool dragging = false;
  int? activePointer;

  @override
  void didUpdateWidget(covariant DraggableCropBlock oldWidget) {
    super.didUpdateWidget(oldWidget);

    if (!dragging) {
      dragOffsetPixels = Offset.zero;
      activePointer = null;
    }
  }

  @override
  Widget build(BuildContext context) {
    final cropName = widget.block.cropName.trim().isEmpty
        ? 'Crop'
        : widget.block.cropName.trim();

    final spacingMeters = CropSpacing.spacingMetersForCrop(cropName);

    final estimatedCount = math.max(
      1,
      CropSpacing.estimatedPlantCount(
        cropName: cropName,
        widthMeters: widget.block.width,
        heightMeters: widget.block.height,
      ),
    );

    final left = widget.block.x * widget.pixelsPerMeter + dragOffsetPixels.dx;
    final top = widget.block.y * widget.pixelsPerMeter + dragOffsetPixels.dy;

    final width = math.max(78.0, widget.block.width * widget.pixelsPerMeter);
    final height = math.max(70.0, widget.block.height * widget.pixelsPerMeter);

    final plantStyle = _PlantIconStyle.forCrop(cropName);

    return Positioned(
      left: left,
      top: top,
      width: width,
      height: height,
      child: Listener(
        behavior: HitTestBehavior.opaque,
        onPointerDown: _onPointerDown,
        onPointerMove: _onPointerMove,
        onPointerUp: _onPointerUp,
        onPointerCancel: _onPointerCancel,
        child: Tooltip(
          message:
              '$cropName\n$estimatedCount planted\n${CropSpacing.spacingLabelForCrop(cropName)}',
          child: AnimatedScale(
            scale: dragging ? 1.035 : 1.0,
            duration: const Duration(milliseconds: 90),
            child: DecoratedBox(
              decoration: BoxDecoration(
                color: Colors.transparent,
                borderRadius: BorderRadius.circular(18),
                border: Border.all(
                  color: dragging ? GardenTheme.good : Colors.transparent,
                  width: dragging ? 2 : 0,
                ),
              ),
              child: ClipRect(
                child: Stack(
                  fit: StackFit.expand,
                  children: [
                    _CropIconBubbleField(
                      cropName: cropName,
                      emoji: plantStyle.emoji,
                      borderColor: plantStyle.borderColor,
                      perennial: plantStyle.perennial,
                      spacingMeters: spacingMeters,
                      pixelsPerMeter: widget.pixelsPerMeter,
                      estimatedCount: estimatedCount,
                    ),
                    Positioned(
                      right: 2,
                      bottom: 2,
                      child: _MapCountBadge(count: estimatedCount),
                    ),
                  ],
                ),
              ),
            ),
          ),
        ),
      ),
    );
  }

  void _onPointerDown(PointerDownEvent event) {
    activePointer = event.pointer;
    dragging = true;
    dragOffsetPixels = Offset.zero;
    setState(() {});
  }

  void _onPointerMove(PointerMoveEvent event) {
    if (activePointer != event.pointer) return;

    setState(() {
      dragOffsetPixels += event.delta;
    });
  }

  void _onPointerUp(PointerUpEvent event) {
    if (activePointer != event.pointer) return;

    final nextX = widget.block.x + dragOffsetPixels.dx / widget.pixelsPerMeter;
    final nextY = widget.block.y + dragOffsetPixels.dy / widget.pixelsPerMeter;

    activePointer = null;
    dragging = false;
    dragOffsetPixels = Offset.zero;
    setState(() {});

    widget.controller.moveCropBlock(
      bedId: widget.bedId,
      blockId: widget.block.id,
      x: nextX,
      y: nextY,
    );
  }

  void _onPointerCancel(PointerCancelEvent event) {
    if (activePointer != event.pointer) return;

    activePointer = null;
    dragging = false;
    dragOffsetPixels = Offset.zero;
    setState(() {});
  }
}

class _CropIconBubbleField extends StatelessWidget {
  const _CropIconBubbleField({

    required this.borderColor,
    required this.perennial,

    required this.pixelsPerMeter,
    required this.estimatedCount,
  });

  final String cropName;
  final String emoji;
  final Color borderColor;
  final bool perennial;
  final double spacingMeters;
  final double pixelsPerMeter;
  final int estimatedCount;

  @override
  Widget build(BuildContext context) {

        final rows = math.max(1, (constraints.maxHeight / 74.0).floor());

        final maxVisible = math.min(estimatedCount, columns * rows);
        final visibleCount = math.max(1, maxVisible);

        return Stack(
          children: [
            for (var index = 0; index < visibleCount; index++)
              _positionedBubble(
                index: index,
                columns: columns,
                rows: rows,
                width: constraints.maxWidth,
                height: constraints.maxHeight,
              ),
          ],
        );
      },
    );
  }

  Widget _positionedBubble({
    required int index,
    required int columns,
    required int rows,
    required double width,
    required double height,
  }) {
    final column = index % columns;
    final row = index ~/ columns;

    final cellWidth = width / columns;
    final cellHeight = height / rows;

    final stagger = row.isOdd ? cellWidth * 0.16 : 0.0;

    final left = (column * cellWidth + cellWidth / 2 - 28 + stagger)
        .clamp(0.0, math.max(0.0, width - 56.0))
        .toDouble();

    final top = (row * cellHeight + cellHeight / 2 - 34)
        .clamp(0.0, math.max(0.0, height - 68.0))
        .toDouble();

    return Positioned(
      left: left,
      top: top,
      width: 56,
      height: 68,
      child: _PlantIconBubble(
        cropName: cropName,
        emoji: emoji,
        borderColor: borderColor,
        perennial: perennial,
      ),
    );

  void transplantSeedlingToBed({
    required String seedlingId,
    required String bedId,
    bool addCropBlockToMap = false,
  }) {
    final seedlingIndex = seedlings.indexWhere(
      (seedling) => seedling.id == seedlingId,
    );

    if (seedlingIndex == -1) return;

    final bedIndex = beds.indexWhere((bed) => bed.id == bedId);
    if (bedIndex == -1) return;

    final seedling = seedlings[seedlingIndex];
    final bed = beds[bedIndex];

    final cropName = seedling.cropName.trim().isEmpty
        ? 'Seedling'
        : seedling.cropName.trim();

    final existingCropNames = bed.crops
        .map((crop) => crop.trim().toLowerCase())
        .where((crop) => crop.isNotEmpty)
        .toSet();

    final updatedCrops = existingCropNames.contains(cropName.toLowerCase())
        ? bed.crops
        : [...bed.crops, cropName];

    final updatedSeedlings = [...seedlings];

    updatedSeedlings[seedlingIndex] = seedling.copyWith(
      bedId: bed.id,
      transplantedAt: DateTime.now(),
    );

    var updatedBed = bed.copyWith(crops: updatedCrops);

    if (addCropBlockToMap) {
      final spacing = CropSpacing.spacingMetersForCrop(cropName);
      final blockWidth = _clampDouble(spacing * 8, spacing, bed.width);
      final blockHeight = _clampDouble(spacing * 2, spacing, bed.height);

      final nextPosition = _nextCropBlockPosition(
        bed: updatedBed,
        cropName: cropName,
        blockWidth: blockWidth,
        blockHeight: blockHeight,
        spacing: spacing,
      );

      final block = CropBlock(
        id: 'block-${DateTime.now().microsecondsSinceEpoch}',
        cropName: cropName,
        x: nextPosition.dx,
        y: nextPosition.dy,
        width: blockWidth,
        height: blockHeight,
      );

      updatedBed = updatedBed.copyWith(
        cropBlocks: [...updatedBed.cropBlocks, block],
      );
    }

    final updatedBeds = [...beds];
    updatedBeds[bedIndex] = updatedBed;

    _recordHistory();

    project = project.copyWith(
      beds: updatedBeds,
      seedlings: updatedSeedlings,
      updatedAt: DateTime.now(),
    );

    selectedBedId = bed.id;

    notifyListeners();
  }

    required double y,

      nextX = _clampDouble(

        project.heightMeters - bed.height,

    final updatedBed = bed.copyWith(x: nextX, y: nextY);

    _recordHistory();

    final updatedBeds = [...beds];

  void addCropRowToBedRect({
    required String bedId,
    required String cropName,
    required double startX,
    required double startY,
    required double endX,
    required double endY,
  }) {
    final targetCrop = cropName.trim();
    if (targetCrop.isEmpty) return;

    final bedIndex = beds.indexWhere((bed) => bed.id == bedId);
    if (bedIndex == -1) return;

    final bed = beds[bedIndex];
    final spacing = CropSpacing.spacingMetersForCrop(targetCrop);

    final existingCropsLower = bed.crops
        .map((crop) => crop.trim().toLowerCase())
        .where((crop) => crop.isNotEmpty)
        .toSet();

    final updatedCrops = existingCropsLower.contains(targetCrop.toLowerCase())
        ? bed.crops
        : [...bed.crops, targetCrop];

    final dx = (endX - startX).abs();
    final dy = (endY - startY).abs();
    final horizontal = dx >= dy;

    final rowThickness = _clampDouble(
      spacing * 2.2,
      _clampDouble(0.65, spacing, horizontal ? bed.height : bed.width),
      horizontal ? bed.height : bed.width,
    );

    final minRowLength = _clampDouble(
      spacing * 6,
      spacing,
      horizontal ? bed.width : bed.height,
    );

    final rawLength = horizontal ? dx : dy;
    final rowLength = _clampDouble(
      rawLength < minRowLength ? minRowLength : rawLength,
      spacing,
      horizontal ? bed.width : bed.height,
    );

    final centerX = ((startX + endX) / 2) - bed.x;
    final centerY = ((startY + endY) / 2) - bed.y;

    late final double blockX;
    late final double blockY;
    late final double blockWidth;
    late final double blockHeight;

    if (horizontal) {
      blockWidth = rowLength;
      blockHeight = rowThickness;

      blockX = _clampDouble(
        centerX - blockWidth / 2,
        0,
        bed.width - blockWidth,
      );

      blockY = _clampDouble(
        centerY - blockHeight / 2,
        0,
        bed.height - blockHeight,
      );
    } else {
      blockWidth = rowThickness;
      blockHeight = rowLength;

      blockX = _clampDouble(
        centerX - blockWidth / 2,
        0,
        bed.width - blockWidth,
      );

      blockY = _clampDouble(
        centerY - blockHeight / 2,
        0,
        bed.height - blockHeight,
      );
    }

    final block = CropBlock(
      id: 'block-${DateTime.now().microsecondsSinceEpoch}',
      cropName: targetCrop,
      x: blockX,
      y: blockY,
      width: blockWidth,
      height: blockHeight,
    );

    _recordHistory();

    final updatedBeds = [...beds];

    updatedBeds[bedIndex] = bed.copyWith(
      crops: updatedCrops,
      cropBlocks: [...bed.cropBlocks, block],
    );

    project = project.copyWith(beds: updatedBeds, updatedAt: DateTime.now());

    selectedBedId = bed.id;

    notifyListeners();
  }

  void _updateSelectedBed(Bed Function(Bed bed) update) {
    if (selectedBedId == null) return;

    final index = beds.indexWhere((bed) => bed.id == selectedBedId);
    if (index == -1) return;

    final currentBed = beds[index];
    final updatedBed = update(currentBed);

    if (updatedBed == currentBed) return;

    if (!allowOverlap && _wouldOverlap(updatedBed, currentBed.id)) return;

    _recordHistory();

  Offset _nextCropPlacementPosition({
    required Bed bed,
    required String cropName,
    required double spacing,
  }) {
    final existingForCrop = bed.cropPlacements
        .where(
          (placement) =>
              placement.cropName.trim().toLowerCase() ==
              cropName.trim().toLowerCase(),
        )
        .length;

    final columnCount = (bed.width / spacing).floor();
    final safeColumnCount = columnCount < 1 ? 1 : columnCount;

    final column = existingForCrop % safeColumnCount;
    final row = existingForCrop ~/ safeColumnCount;

    final rawX = spacing / 2 + column * spacing;
    final rawY = spacing / 2 + row * spacing;

    return Offset(
      _clampDouble(rawX, spacing / 2, bed.width - spacing / 2),
      _clampDouble(rawY, spacing / 2, bed.height - spacing / 2),
    );
  }

  Offset _nextCropBlockPosition({
    required Bed bed,
    required String cropName,
    required double blockWidth,
    required double blockHeight,
    required double spacing,
  }) {
    final existingForCrop = bed.cropBlocks
        .where(
          (block) =>
              block.cropName.trim().toLowerCase() ==
              cropName.trim().toLowerCase(),
        )
        .length;

    final availableWidth = bed.width - blockWidth;
    final availableHeight = bed.height - blockHeight;

    final columnCount = availableWidth <= 0
        ? 1
        : ((availableWidth / spacing).floor() + 1).clamp(1, 9999);

    final column = existingForCrop % columnCount;
    final row = existingForCrop ~/ columnCount;

    final rawX = column * spacing;
    final rawY = row * spacing;

    return Offset(
      _clampDouble(rawX, 0, availableWidth),
      _clampDouble(rawY, 0, availableHeight),
    );
  }

  double _snapCropCoordinate({
    required double value,
    required double spacing,
    required double max,
  }) {
    if (max <= 0) return 0;
    if (spacing <= 0) return _clampDouble(value, 0, max);

    final halfSpacing = spacing / 2;

    if (max <= halfSpacing) {
      return max / 2;
    }

    final snapped =
        ((value - halfSpacing) / spacing).round() * spacing + halfSpacing;

    return _clampDouble(snapped, halfSpacing, max - halfSpacing);
  }

  double _snapCropBlockCoordinate({
    required double value,
    required double spacing,
    required double min,
    required double max,
  }) {
    if (max < min) return min;
    if (spacing <= 0) return _clampDouble(value, min, max);

    final snapped = (value / spacing).round() * spacing;

    return _clampDouble(snapped, min, max);
  }

  double _snapCropBlockSize({
    required double value,
    required double spacing,
    required double min,
    required double max,
  }) {
    if (max < min) return min;
    if (spacing <= 0) return _clampDouble(value, min, max);

    final snapped = (value / spacing).round() * spacing;

    return _clampDouble(snapped, min, max);
  }

  bool _wouldOverlap(Bed candidate, String candidateId) {
    for (final other in beds) {
      if (other.id == candidateId) continue;

      if (_bedsOverlap(candidate, other)) {
        return true;
      }
    }

    return false;
  }

  bool _bedsOverlap(Bed a, Bed b) {
    final leftA = a.x;
    final rightA = a.x + a.width;
    final topA = a.y;
    final bottomA = a.y + a.height;

    final leftB = b.x;
    final rightB = b.x + b.width;
    final topB = b.y;
    final bottomB = b.y + b.height;

    return leftA < rightB && rightA > leftB && topA < bottomB && bottomA > topB;
  }

  void _recordHistory() {
    _undoStack.add(project);
    _redoStack.clear();

    const maxHistory = 50;

    if (_undoStack.length > maxHistory) {
      _undoStack.removeAt(0);
    }
  }

  double _snap(double value, double step) {
    return (value / step).round() * step;
  }

  double _clampDouble(double value, double min, double max) {
    if (max < min) return min;
    return value.clamp(min, max).toDouble();
  }
}

class CropBlock {
  const CropBlock({
    required this.id,
    required this.cropName,
    required this.x,
    required this.y,
    required this.width,
    required this.height,
  });

  final String id;
  final String cropName;
  final double x;
  final double y;
  final double width;
  final double height;

  CropBlock copyWith({
    String? id,
    String? cropName,
    double? x,
    double? y,
    double? width,
    double? height,
  }) {
    return CropBlock(
      id: id ?? this.id,
      cropName: cropName ?? this.cropName,
      x: x ?? this.x,
      y: y ?? this.y,
      width: width ?? this.width,
      height: height ?? this.height,
    );
  }

  Map<String, dynamic> toJson() {
    return {
      'id': id,
      'cropName': cropName,
      'x': x,
      'y': y,
      'width': width,
      'height': height,
    };
  }

  factory CropBlock.fromJson(Map<String, dynamic> json) {
    return CropBlock(
      id: json['id']?.toString() ?? '',
      cropName: json['cropName']?.toString() ?? '',
      x: _readDouble(json['x']),
      y: _readDouble(json['y']),
      width: _readDouble(json['width'], fallback: 1),
      height: _readDouble(json['height'], fallback: 0.5),
    );
  }

  static double _readDouble(Object? value, {double fallback = 0}) {
    if (value is num) return value.toDouble();
    if (value is String) return double.tryParse(value) ?? fallback;
    return fallback;
  }
}

class CropSpacing {
  const CropSpacing._();

  static double spacingMetersForCrop(String cropName) {
    final crop = cropName.trim().toLowerCase();

    if (crop.contains('strawberry')) return 0.30;
    if (crop.contains('raspberry')) return 0.45;
    if (crop.contains('blueberry')) return 0.75;

    if (crop.contains('lettuce')) return 0.25;
    if (crop.contains('spinach')) return 0.20;
    if (crop.contains('kale')) return 0.45;
    if (crop.contains('broccoli')) return 0.45;
    if (crop.contains('cabbage')) return 0.45;

    if (crop.contains('tomato')) return 0.60;
    if (crop.contains('pepper')) return 0.45;
    if (crop.contains('cucumber')) return 0.45;
    if (crop.contains('zucchini')) return 0.90;
    if (crop.contains('squash')) return 0.90;

    if (crop.contains('carrot')) return 0.08;
    if (crop.contains('radish')) return 0.08;
    if (crop.contains('beet')) return 0.10;
    if (crop.contains('onion')) return 0.10;
    if (crop.contains('garlic')) return 0.15;

    if (crop.contains('basil')) return 0.25;
    if (crop.contains('parsley')) return 0.20;
    if (crop.contains('cilantro')) return 0.15;
    if (crop.contains('thyme')) return 0.25;
    if (crop.contains('mint')) return 0.30;

    return 0.30;
  }

  static String spacingLabelForCrop(String cropName) {
    final spacing = spacingMetersForCrop(cropName);
    final centimeters = (spacing * 100).round();

    return '${centimeters}cm spacing';
  }

  static int estimatedPlantCount({
    required String cropName,
    required double widthMeters,
    required double heightMeters,
  }) {
    final spacing = spacingMetersForCrop(cropName);

    if (widthMeters <= 0 || heightMeters <= 0 || spacing <= 0) {
      return 0;
    }

    final columns = (widthMeters / spacing).floor();
    final rows = (heightMeters / spacing).floor();

    return columns * rows;
  }
}
