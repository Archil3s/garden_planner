import net.runelite.api.coords.WorldPoint;

public final class TilePoint {
    public final int x;
    public final int y;
    public final int plane;

    public TilePoint(int x, int y, int plane) {
        this.x = x;
        this.y = y;
        this.plane = plane;
    }

    public static TilePoint of(WorldPoint wp) {
        if (wp == null) {
            return null;
        }
        return new TilePoint(wp.getX(), wp.getY(), wp.getPlane());
    }

    public WorldPoint toWorldPoint() {
        return new WorldPoint(x, y, plane);
    }

    public int chebyshevDistance(TilePoint other) {
        if (other == null || plane != other.plane) {
            return Integer.MAX_VALUE;
        }
        return Math.max(Math.abs(x - other.x), Math.abs(y - other.y));
    }

    public TilePoint stepToward(TilePoint destination) {
        if (destination == null || plane != destination.plane) {
            return this;
        }

        int dx = Integer.compare(destination.x, x);
        int dy = Integer.compare(destination.y, y);

        return new TilePoint(x + dx, y + dy, plane);
    }

    @Override
    public String toString() {
        return "TilePoint{x=" + x + ", y=" + y + ", plane=" + plane + "}";
    }
}

        this.inRange = inRange;
        this.improvesDanceDistance = improvesDanceDistance;
        this.score = score;

    public static TileScore score(TilePoint tile,
                                  TilePoint target,
                                  int currentDistance,
                                  boolean reachable,
                                  boolean pathable,
                                  boolean lineOfSight,
                                  int weaponRange) {
        int distance = tile == null ? Integer.MAX_VALUE : tile.chebyshevDistance(target);
        boolean inRange = distance <= Math.max(1, weaponRange);
        boolean improvesDanceDistance = distance >= currentDistance && distance <= currentDistance + 2;

        int s = 0;

        if (reachable) {
            s += 25;
        }
        if (pathable) {
            s += 25;
        }
        if (lineOfSight) {
            s += 20;
        }
        if (inRange) {
            s += 20;
        }
        if (improvesDanceDistance) {
            s += 20;
        }

        if (!reachable || !pathable) {
            s -= 45;
        }
        if (!lineOfSight) {
            s -= 20;
        }
        if (distance == 0) {
            s -= 30;
        }

        return new TileScore(
                tile,
                distance,
                reachable,
                pathable,
                lineOfSight,
                inRange,
                improvesDanceDistance,
                s
        );
    }

$ScenePointUtilJava = @"
$P

import net.runelite.api.Tile;
import net.runelite.api.coords.WorldPoint;

import java.lang.reflect.Method;
import java.util.ArrayList;
import java.util.Collections;
import java.util.List;

public final class ScenePointUtil {
    private ScenePointUtil() {
    }

    public static List<WorldPoint> toWorldPoints(Object value) {
        if (value == null) {
            return Collections.emptyList();
        }

        if (value instanceof WorldPoint) {
            List<WorldPoint> single = new ArrayList<>();
            single.add((WorldPoint) value);
            return single;
        }

        if (!(value instanceof List<?>)) {
            return Collections.emptyList();
        }

        List<WorldPoint> result = new ArrayList<>();

        for (Object item : (List<?>) value) {
            WorldPoint point = toWorldPoint(item);
            if (point != null) {
                result.add(point);
            }
        }

        return result;
    }

    public static WorldPoint toWorldPoint(Object item) {
        if (item == null) {
            return null;
        }

        if (item instanceof WorldPoint) {
            return (WorldPoint) item;
        }

        if (item instanceof Tile) {
            return ((Tile) item).getWorldLocation();
        }

        try {
            Method method = item.getClass().getMethod("getWorldLocation");
            Object value = method.invoke(item);
            return value instanceof WorldPoint ? (WorldPoint) value : null;
        } catch (Exception ignored) {
            return null;
        }
    }
}

$MovementPredictorJava = @"
$P

import com.tonic.api.MovementAPI;
import com.tonic.api.SceneAPI;
import net.runelite.api.coords.WorldPoint;

import java.util.List;

public final class MovementPredictor {
    public MovementSnapshot snapshot(WorldPoint currentWorldPoint) {
        TilePoint current = TilePoint.of(currentWorldPoint);

        if (currentWorldPoint == null) {
            return MovementSnapshot.stationary(null);
        }

        boolean moving = MovementAPI.isMoving();
        WorldPoint destinationWorldPoint = MovementAPI.getDestinationWorldPoint();

        if (!moving || destinationWorldPoint == null) {
            return MovementSnapshot.stationary(current);
        }

        TilePoint destination = TilePoint.of(destinationWorldPoint);
        boolean pathable = MovementAPI.canPathTo(destinationWorldPoint);
        boolean reachable = SceneAPI.isReachable(currentWorldPoint, destinationWorldPoint);

        Object pathObject = SceneAPI.checkPointsTo(currentWorldPoint, destinationWorldPoint);
        List<WorldPoint> path = ScenePointUtil.toWorldPoints(pathObject);

        TilePoint predictedNext = predictNextTile(current, destination, path);

        return new MovementSnapshot(
                current,
                destination,
                predictedNext,
                true,
                pathable,
                reachable,
                path == null ? 0 : path.size()
        );
    }

    private TilePoint predictNextTile(TilePoint current, TilePoint destination, List<WorldPoint> path) {
        if (current == null) {
            return null;
        }

        if (path != null) {
            for (WorldPoint point : path) {
                TilePoint tile = TilePoint.of(point);
                if (tile != null && tile.chebyshevDistance(current) > 0) {
                    return tile;
                }
            }
        }

        if (destination == null) {
            return current;
        }

        return current.stepToward(destination);
    }
}

$TickDanceModelJava = @"
$P

public final class TickDanceModel {
    public TickDanceEstimate estimate(MovementSnapshot movement,
                                      TilePoint targetTile,
                                      int weaponRange) {
        if (movement == null || movement.currentTile == null || targetTile == null) {
            return new TickDanceEstimate(
                    TickDanceStatus.UNKNOWN,
                    movement == null ? null : movement.currentTile,
                    movement == null ? null : movement.destinationTile,
                    movement == null ? null : movement.predictedNextTile,
                    Integer.MAX_VALUE,
                    Integer.MAX_VALUE,
                    0,
                    "Missing movement or target tile"
            );
        }

        int distanceNow = movement.currentTile.chebyshevDistance(targetTile);
        TilePoint next = movement.predictedNextTile == null ? movement.currentTile : movement.predictedNextTile;
        int distanceNext = next.chebyshevDistance(targetTile);
        int range = Math.max(1, weaponRange);

        if (!movement.moving) {
            return new TickDanceEstimate(
                    TickDanceStatus.NOT_MOVING,
                    movement.currentTile,
                    movement.destinationTile,
                    next,
                    distanceNow,
                    distanceNext,
                    85,
                    "Local player is not moving"
            );
        }

        if (!movement.pathableToDestination || !movement.reachableDestination) {
            return new TickDanceEstimate(
                    TickDanceStatus.STALLED_OR_BLOCKED,
                    movement.currentTile,
                    movement.destinationTile,
                    next,
                    distanceNow,
                    distanceNext,
                    70,
                    "Destination is not pathable or not reachable"
            );
        }

        if (distanceNext <= range && distanceNow > range) {
            return new TickDanceEstimate(
                    TickDanceStatus.MOVING_TO_RANGE,
                    movement.currentTile,
                    movement.destinationTile,
                    next,
                    distanceNow,
                    distanceNext,
                    80,
                    "Next tile moves into attack range"
            );
        }

        if (distanceNext > distanceNow) {
            return new TickDanceEstimate(
                    TickDanceStatus.MOVING_AWAY,
                    movement.currentTile,
                    movement.destinationTile,
                    next,
                    distanceNow,
                    distanceNext,
                    75,
                    "Next tile increases distance from target"
            );
        }

        if (distanceNext == distanceNow && distanceNow <= range + 1) {
            return new TickDanceEstimate(
                    TickDanceStatus.ORBITING_TARGET,
                    movement.currentTile,
                    movement.destinationTile,
                    next,
                    distanceNow,
                    distanceNext,
                    75,
                    "Next tile preserves close distance around target"
            );
        }

        if (distanceNext < distanceNow && distanceNext == 0) {
            return new TickDanceEstimate(
                    TickDanceStatus.CROSSING_TARGET,
                    movement.currentTile,
                    movement.destinationTile,
                    next,
                    distanceNow,
                    distanceNext,
                    60,
                    "Next tile overlaps target tile"
            );
        }

        return new TickDanceEstimate(
                TickDanceStatus.UNKNOWN,
                movement.currentTile,
                movement.destinationTile,
                next,
                distanceNow,
                distanceNext,
                45,
                "Movement does not match a clean tick-dance category"
        );
    }
}

import javax.inject.Inject;
import java.awt.Color;
import java.awt.Dimension;
import java.awt.Font;
import java.awt.Graphics2D;
import java.awt.Polygon;

public class TheoreticalPidOverlay extends Overlay {
    private static final Color GOOD = new Color(0, 170, 0, 110);
    private static final Color WARN = new Color(220, 180, 0, 110);
    private static final Color BAD = new Color(200, 0, 0, 110);
    private static final Color NEXT = new Color(0, 140, 255, 150);
    private static final Color DEST = new Color(180, 0, 255, 130);
    private static final Color TEXT_SHADOW = Color.BLACK;

    private final Client client;
    private final TheoreticalPidPlugin plugin;
    private final TheoreticalPidConfig config;

    @Inject
    public TheoreticalPidOverlay(Client client, TheoreticalPidPlugin plugin, TheoreticalPidConfig config) {
        this.client = client;
        this.plugin = plugin;
        this.config = config;
        setPosition(OverlayPosition.DYNAMIC);
        setLayer(OverlayLayer.ABOVE_SCENE);
        setPriority(OverlayPriority.LOW);
    }

    @Override
    public Dimension render(Graphics2D graphics) {
        OverlayState state = plugin.getOverlayState();

        if (state == null) {
            return null;
        }

        if (config.showTileOverlay()) {
            for (TileScore score : state.tiles) {
                renderTile(graphics, score.tile == null ? null : score.tile.toWorldPoint(), pickColor(score));
            }

            if (state.movement != null) {
                if (state.movement.destinationTile != null) {
                    renderTile(graphics, state.movement.destinationTile.toWorldPoint(), DEST);
                }

                if (state.movement.predictedNextTile != null) {
                    renderTile(graphics, state.movement.predictedNextTile.toWorldPoint(), NEXT);
                }
            }
        }

        if (config.showInfoPanel()) {
            renderInfo(graphics, state);
        }

        return null;
    }

    private Color pickColor(TileScore score) {
        if (score.reachable && score.pathable && score.lineOfSight && score.inRange) {
            return GOOD;
        }

        if (score.reachable || score.pathable) {
            return WARN;
        }

        return BAD;
    }

    private void renderTile(Graphics2D graphics, WorldPoint worldPoint, Color fill) {
        if (worldPoint == null) {
            return;
        }

        LocalPoint localPoint = LocalPoint.fromWorld(client, worldPoint);

        if (localPoint == null) {
            return;
        }

        Polygon polygon = Perspective.getCanvasTilePoly(client, localPoint);

        if (polygon == null) {
            return;
        }

        graphics.setColor(fill);
        graphics.fill(polygon);
        graphics.setColor(fill.darker());
        graphics.draw(polygon);
    }

    private void renderInfo(Graphics2D graphics, OverlayState state) {
        int x = 10;
        int y = 36;
        int h = 14;

        graphics.setFont(new Font("Arial", Font.PLAIN, 11));

        drawLine(graphics, x, y, "Validity: " + state.validity, Color.WHITE);
        drawLine(graphics, x, y + h, "Target: " + state.targetName, Color.WHITE);
        drawLine(graphics, x, y + h * 2, "Wildy: " + state.wildyLevel + " | HP: " + state.hitpoints + " | Spec: " + state.specEnergy, Color.WHITE);
        drawLine(graphics, x, y + h * 3, "Moving: " + state.localMoving + " | Spec enabled: " + state.specialEnabled, Color.WHITE);

        if (state.tickDance != null) {
            drawLine(graphics, x, y + h * 4, "Tick dance: " + state.tickDance.status + " @" + state.tickDance.confidence + "%", Color.WHITE);
            drawLine(graphics, x, y + h * 5, "Dist now/next: " + state.tickDance.distanceNow + " -> " + state.tickDance.distanceNext, Color.WHITE);
            drawLine(graphics, x, y + h * 6, "Reason: " + state.tickDance.reason, Color.WHITE);
        }

        if (state.movement != null) {
            drawLine(graphics, x, y + h * 7, "Dest: " + state.movement.destinationTile + " | Path len: " + state.movement.pathLengthToDestination, Color.WHITE);
        }
    }

    private void drawLine(Graphics2D graphics, int x, int y, String text, Color color) {
        graphics.setColor(TEXT_SHADOW);
        graphics.drawString(text, x + 1, y + 1);
        graphics.setColor(color);
        graphics.drawString(text, x, y);
    }

        if (validity != FightValidity.VALID_ONE_V_ONE) {
            overlayState = stateForInvalid(validity, local, target);
            return;
        }

        WorldPoint localPoint = local.getWorldLocation();
        WorldPoint targetPoint = target.getWorldLocation();

        MovementSnapshot movement = movementPredictor.snapshot(localPoint);
        TickDanceEstimate tickDance = tickDanceModel.estimate(
                movement,
                TilePoint.of(targetPoint),
                config.assumedWeaponRange()
        );

        overlayState = new OverlayState(
                validity,
                TilePoint.of(localPoint),
                TilePoint.of(targetPoint),
                movement,
                tickDance,
                MovementAPI.isMoving(),
                CombatAPI.isSpecEnabled(),
                CombatAPI.isPoisoned() || CombatAPI.isVenomed(),
                CombatAPI.getNextPoisonDamage(),
                GameAPI.getWildyLevel(),
                CombatAPI.getHealth(),
                CombatAPI.getSpecEnergy(),
                target.getName(),
                scoreTiles(localPoint, targetPoint)
        );
    }

    private Player getLocalPlayer() {
        PlayerEx localEx = PlayerAPI.getLocal();

        if (localEx != null && localEx.getPlayer() != null) {
            return localEx.getPlayer();
        }

        return client == null ? null : client.getLocalPlayer();
    }

    private Actor getCombatTarget(Player local) {
        Object raw = ActorAPI.getInCombatWith();
        Actor converted = toActor(raw);

        if (converted != null) {
            return converted;
        }

        return local == null ? null : local.getInteracting();
    }

    private Actor toActor(Object raw) {
        if (raw == null) {
            return null;
        }

        if (raw instanceof Actor) {
            return (Actor) raw;
        }

        Actor actor = callActorGetter(raw, "getActor");
        if (actor != null) {
            return actor;
        }

        actor = callActorGetter(raw, "getPlayer");
        if (actor != null) {
            return actor;
        }

        return callActorGetter(raw, "getNpc");
    }

    private Actor callActorGetter(Object target, String methodName) {
        try {
            Method method = target.getClass().getMethod(methodName);
            Object value = method.invoke(target);
            return value instanceof Actor ? (Actor) value : null;
        } catch (Exception ignored) {
            return null;
        }
    }

    private FightValidity assessFightValidity(Player local, Actor target) {
        if (local == null) {
            return FightValidity.NO_LOCAL_PLAYER;
        }

        if (target == null) {
            return FightValidity.NO_TARGET;
        }

        if (!CombatAPI.isInPvpArea() && GameAPI.getWildyLevel() <= 0) {
            return FightValidity.NOT_IN_PVP;
        }

        if (CombatAPI.inMultiWay()) {
            return FightValidity.MULTIWAY;
        }

        WorldPoint localPoint = local.getWorldLocation();
        WorldPoint targetPoint = target.getWorldLocation();

        if (localPoint == null || targetPoint == null || localPoint.getPlane() != targetPoint.getPlane()) {
            return FightValidity.DIFFERENT_PLANE;
        }

        return FightValidity.VALID_ONE_V_ONE;
    }

    private OverlayState stateForInvalid(FightValidity validity, Player local, Actor target) {
        MovementSnapshot movement = local == null ? null : movementPredictor.snapshot(local.getWorldLocation());

        return new OverlayState(
                validity,
                local == null ? null : TilePoint.of(local.getWorldLocation()),
                target == null ? null : TilePoint.of(target.getWorldLocation()),
                movement,
                null,
                MovementAPI.isMoving(),
                CombatAPI.isSpecEnabled(),
                CombatAPI.isPoisoned() || CombatAPI.isVenomed(),
                CombatAPI.getNextPoisonDamage(),
                GameAPI.getWildyLevel(),
                CombatAPI.getHealth(),
                CombatAPI.getSpecEnergy(),
                target == null ? "?" : target.getName(),
                Collections.emptyList()
        );
    }

    private List<TileScore> scoreTiles(WorldPoint localPoint, WorldPoint targetPoint) {
        if (localPoint == null || targetPoint == null) {
            return Collections.emptyList();
        }

        List<WorldPoint> candidates = surrounding(localPoint, config.overlayRadius());
        List<TileScore> result = new ArrayList<>();

        TilePoint localTile = TilePoint.of(localPoint);
        TilePoint targetTile = TilePoint.of(targetPoint);
        int currentDistance = localTile.chebyshevDistance(targetTile);

        for (WorldPoint candidate : candidates) {
            boolean reachable = SceneAPI.isReachable(localPoint, candidate);
            boolean pathable = MovementAPI.canPathTo(candidate);
            boolean los = SceneAPI.hasLineOfSightTo(candidate, targetPoint);

            result.add(TileScore.score(
                    TilePoint.of(candidate),
                    targetTile,
                    currentDistance,
                    reachable,
                    pathable,
                    los,
                    config.assumedWeaponRange()
            ));
        }

        result.sort((a, b) -> Integer.compare(b.score, a.score));
        return result;
    }

    private List<WorldPoint> surrounding(WorldPoint center, int radius) {
        if (center == null) {
            return Collections.emptyList();
        }

        int r = Math.max(1, Math.min(4, radius));
        List<WorldPoint> points = new ArrayList<>();

        for (int dx = -r; dx <= r; dx++) {
            for (int dy = -r; dy <= r; dy++) {
                points.add(new WorldPoint(center.getX() + dx, center.getY() + dy, center.getPlane()));
            }
        }

        return points;
    }
